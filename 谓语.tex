\subsection{语态}
\begin{subexample} \ \\
		Jack \bl[谓语]{repairs} the bike everyday.\\
		The bike \bl[时态]{is} \bl[被动]{repaired} by Jack everyday.
\end{subexample}
\begin{subexample}\ \\
	Jack \bl[谓语]{repaired} bike yesterday.\\
	The bike \bl[时态]{was} \bl[被动]{repaired} by Jack.
\end{subexample}
\begin{subexample}
	Jack \bl[谓语]{is repairing} the bike now.\\
	The bike \bl[时态]{is being} \bl[被动]{repaired} by Jack now.
\end{subexample}
\begin{subexample}
	Jack \bl[谓语]{has repaired} his bike.\\
	His bike \bl[时态]{has been} \bl[被动]{repaired} by Jack.
\end{subexample}
\begin{subexample}
	Jack \bl[谓语]{built} the bridge last year.\\
	The bridge \bl[时态]{was} \bl[被动]{built} by last year.
\end{subexample}
