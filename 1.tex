\section{定语}
\begin{definition}
	定语:句子中，一切\uline{修订限定}n.(名词)的词或结构
\end{definition}
\begin{example}
	The \bl{ugly}(定语) dog bit \bl{handsome} Steven \bl{who helped me last week}.
\end{example}
\begin{example}
	There is a boy asking barber cut his hair in David Beckham's style.
\end{example}
\begin{example}
	A \bl{dancing} woman.
\end{example}
\begin{example}
		A woman \bl{picking apples}.
\end{example}
\begin{example}
	This is a bridge \bl{founded by Jack}.
\end{example}
\begin{example}
	A \bl{repaired} bicycle.
\end{example}
\begin{example}
	Bicycle \bl{repaired by Losy}.
\end{example}
\begin{tabular}{rcll}
	ugly&adj.&定语&\\
	running&r$\rightarrow$adj&定语&(现在分词)主/进\\
	killed&v$\rightarrow$adj&定语&(过去分词)被/完\\
	dying&v$\rightarrow$adj&定语&(现在分词)主/进\\
	died&v$\rightarrow$adj&定语&(过去分词)被/完
\end{tabular}
\\
\begin{enumerate}
	\item adj.(ugly)dog\\adj$\rightarrow$定
	\item (running)dog\\v$\rightarrow$adj\\现在分词$\rightarrow$定
	\item (killed) dog\\v$\rightarrow$adj\\过去分词$\rightarrow$定
	\item dog (to save)\\to do\\不定式
	\item dog (of great ugliness)\\prep+n\\介宾短语.
	\item dog (which is) killed by Steven\\定语从句
\end{enumerate}
\section{状语}
\begin{definition}
	状语：句子中，一切说明动作(v.)\dotuline{发生状态}词或结构。
\end{definition}
\begin{example}
	adv.\ The dog \bl{crazily} bit Steven.\\
	形态：adv.$\rightarrow$种类：方式状语
\end{example}
\begin{example}
	\bl{Seeing Steven}, she ran away.\\形态：现在分词$\rightarrow$种类：原因状语
\end{example}
\begin{example}
	\bl{Enraged(very angry) by Steven}, She cried \bl{all night}.\\
	形态：过去分词$\rightarrow$种类：原因状语
\end{example}
\begin{example}
	He apent all his savings \bl{to save me}.\\
	形态：不定式$\rightarrow$种类:目的状语
\end{example}
\begin{example}
	The dog \bl{crazily} bit Steven \bl{in the park}.\\
	形态：介宾短语$\rightarrow$种类：地点状语
\end{example}
\begin{example}
	He saved me \bl{by spending all his savings}.\\
	形态：介宾短语$\rightarrow$种类：方式状语
\end{example}
\begin{example}
	\bl{Because she saw Steven}, she ran away.\\
	conj.\ \ 状语\dotuline{从句}
\end{example}
\begin{example}
	\bl{If we are united}, we stand. $\rightarrow$\bl{United}, we stand.
\end{example}
